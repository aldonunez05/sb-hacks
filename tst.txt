Create a script to seed challenges
Create scripts/seedChallenges.js:
javascriptconst mongoose = require('mongoose');
require('dotenv').config();

const Challenge = require('../src/models/Challenge');

const challenges = [
  { prompt: 'Take a photo with your best friend', category: 'social' },
  { prompt: 'Show us your lunch', category: 'food' },
  { prompt: 'Capture your current view', category: 'nature' },
  { prompt: 'Photo of something blue', category: 'creative' },
  { prompt: 'Your favorite spot on campus', category: 'nature' },
  { prompt: 'Something that made you smile today', category: 'random' },
  { prompt: 'Your study setup', category: 'random' },
  { prompt: 'A random act of kindness', category: 'social' },
  { prompt: 'Show us your breakfast', category: 'food' },
  { prompt: 'Take a selfie with a stranger', category: 'social' },
];

async function seedChallenges() {
  try {
    await mongoose.connect(process.env.MONGODB_URI);
    console.log('Connected to MongoDB');

    await Challenge.deleteMany({});
    await Challenge.insertMany(challenges);

    console.log('âœ… Challenges seeded successfully!');
    process.exit(0);
  } catch (error) {
    console.error('Error seeding challenges:', error);
    process.exit(1);
  }
}

seedChallenges();

Common Issues & Fixes
"Cannot connect to MongoDB"

Make sure MongoDB is running: brew services start mongodb-community
Or use MongoDB Atlas cloud URL

"localhost not working on phone"

Change API_URL in client .env to your computer's IP
Find IP: ifconfig (Mac/Linux) or ipconfig (Windows)
Example: EXPO_PUBLIC_API_URL=http://192.168.1.100:3000/api

"Camera permissions denied"

Delete app from phone and reinstall
Or go to phone Settings â†’ Apps â†’ Expo Go â†’ Permissions

"Push notifications not working"

Make sure you're testing on a real device (not simulator)
Ensure Firebase credentials are correct
Check that push token is being saved to database
